SQL Setup
Run in Supabase SQL Editor → Database → SQL Editor → New Query
CREATE TABLE expenses (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  date text, category text, subcategory text,
  supplier text, amount numeric, vat_rate integer,
  card text, card_notes text, notes text,
  justification text, necessity text,
  track_roi boolean DEFAULT false,
  sf_done boolean DEFAULT false,
  checked boolean DEFAULT false,
  review_status text DEFAULT '',
  month_key text, created_at timestamptz DEFAULT now()
);
CREATE TABLE recurring (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text, category text, amount numeric,
  vat_rate integer, card text, notes text, justification text,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE recurring_confirmations (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  recurring_id uuid, month_key text,
  confirmed boolean DEFAULT false, actual_amount numeric
);
CREATE TABLE providers (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text, cost numeric, revenue numeric,
  notes text, month_key text, created_at timestamptz DEFAULT now()
);
CREATE TABLE roi_items (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name text, type text, monthly_cost numeric,
  value_return numeric, target_roi numeric,
  notes text, month_key text, created_at timestamptz DEFAULT now()
);
-- postage tally (stored as JSON blob per month, no separate table needed)
